class SwitchControlStructure{    
         public static void main(String[] args){
	int letter='A';
              switch(letter){
                     case 'a':
                           System.out.println("Lowercase Letter"); 
                            break;
                     case 'A':
                            System.out.println("Uppercase Letter"); 
                            break;
              
           default:
                   System.out.println("Invalid Letter"); 
                   break;
              } 
       }
